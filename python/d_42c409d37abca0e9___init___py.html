11# *** For developers *** 

12# 

13# When adding new options, update both the file config_default.yaml 

14# and Config._TEMPLATE. 

15# If the template is not updated, new options will simply be ignored. 

16 

17from functools import lru_cache 

18from pathlib import Path 

19from typing import Any, Iterable, Tuple 

20 

21import confuse 

22 

23 

24class Config: 

25 """ 

26 Runtime configuration parameters. 

27 

28 Provides dict-like access to configuration parameters. 

29 Modifications apply to the current process only and do not 

30 modify any configuration files. 

31 

32 See https://scipp.github.io/reference/runtime-configuration.html 

33 """ 

34 

35 _TEMPLATE = { 

36 'colors': confuse.MappingValues(str), 

37 'plot': { 

38 'aspect': confuse.OneOf((confuse.Number(), str)), 

39 'color': confuse.Sequence(str), 

40 'dpi': int, 

41 'height': int, 

42 'linestyle': confuse.Sequence(confuse.Optional(str)), 

43 'linewidth': confuse.Sequence(int), 

44 'marker': confuse.Sequence(str), 

45 'bounding_box': confuse.Sequence(confuse.Number()), 

46 'params': { 

47 'cbar': bool, 

48 'cmap': str, 

49 'color': confuse.Optional(str), 

50 'nan_color': str, 

51 'norm': confuse.Choice(('linear', 'log')), 

52 'over_color': str, 

53 'under_color': str, 

54 'vmax': confuse.Optional(confuse.Number()), 

55 'vmin': confuse.Optional(confuse.Number()), 

56 }, 

57 'pixel_ratio': confuse.Number(), 

58 'width': confuse.Number(), 

59 }, 

60 'table_max_size': int, 

61 } 

62 

63 def __init__(self): 

64 self._cfg = confuse.LazyConfig('scipp', __name__) 

65 self._cfg.set( 

66 confuse.YamlSource('./scipp.config.yaml', 

67 optional=True, 

68 loader=self._cfg.loader)) 

69 

70 def config_dir(self) -> Path: 

71 """ 

72 Return the directory for configuration files. 

73 

74 The directory is created if it does not already exist. 

75 """ 

76 return Path(self._cfg.config_dir()) 

77 

78 def config_path(self) -> Path: 

79 """ 

80 Return the path to the configuration file. 

81 

82 The file may not exist but its folder is created if it does not already exist. 

83 """ 

84 return Path(self._cfg.user_config_path()) 

85 

86 @lru_cache() 

87 def get(self) -> dict: 

88 """Return parameters as a dict.""" 

89 try: 

90 self._cfg.read(user=True, defaults=True) 

91 except PermissionError: 

92 # On some systems, the user configuration directories are read-only. 

93 # confuse tries to create a subdirectory for scipp if it does not 

94 # already exist and that raises PermissionError. 

95 # Fall back to the default configuration shipped as part of the 

96 # source code in this case. 

97 self._cfg.read(user=False, defaults=True) 

98 return self._cfg.get(self._TEMPLATE) 

99 

100 def __getitem__(self, name: str): 

101 """Return parameter of given name.""" 

102 return self.get()[name] 

103 

104 def __setitem__(self, name: str, value): 

105 """Change the value of a parameter.""" 

106 if name not in self.get(): 

107 raise TypeError( 

108 f"New items cannot be inserted into the configuration, got '{name}'.") 

109 self.get()[name] = value 

110 

111 def keys(self) -> Iterable[str]: 

112 """Returns iterable over parameter names.""" 

113 yield from self.get().keys() 

114 

115 def values(self) -> Iterable[Any]: 

116 """Returns iterable over parameter values.""" 

117 yield from self.get().values() 

118 

119 def items(self) -> Iterable[Tuple[str, Any]]: 

120 """Returns iterable over parameter names and values.""" 

121 yield from self.get().items()
