134 def set_position_params(self, params): 

135 limits = params.limits 

136 center = params.center 

137 box_size = params.box_size 

138 

139 if self._pixel_size is None: 

140 # Note the value of 0.05 is arbitrary here. It is a sensible 

141 # guess to render a plot that is not too crowded and shows 

142 # individual pixels. 

143 self._pixel_size = 0.05 * np.mean(box_size) 

144 box_size += self._pixel_size 

145 

146 self._create_outline(limits=limits, box_size=box_size, center=center) 

147 

148 self.axticks = self._generate_axis_ticks_and_labels( 

149 box_size=box_size, limits=limits, components=params.components) 

150 

151 self.point_cloud_positions = params.positions.values.astype('float32') 

152 self.point_clouds["main"] = self._create_point_cloud( 

153 positions=self.point_cloud_positions) 

154 

155 # Set camera controller target 

156 distance_from_center = 1.2 

157 camera_position = list(np.array(center) + distance_from_center * box_size) 

158 camera_lookat = tuple(center) 

159 if self.initial_camera_view is not None: 

160 camera_position = self.initial_camera_view.get('position', camera_position) 

161 camera_lookat = tuple(self.initial_camera_view.get( 

162 'look_at', camera_lookat)) 

163 self.initial_camera_view = None 

164 self.camera.position = camera_position 

165 cam_pos_norm = np.linalg.norm(self.camera.position) 

166 box_mean_size = np.linalg.norm(box_size) 

167 self.camera.near = 0.01 * box_mean_size 

168 self.camera.far = 5.0 * cam_pos_norm 

169 self.controls.target = camera_lookat 

170 self.camera.lookAt(camera_lookat) 

171 # TODO: Update OrbitControls maxDistance. This should be removed once 

172 # https://github.com/jupyter-widgets/pythreejs/issues/366 is resolved. 

173 self.controls.maxDistance = self.camera.far * 5.0 

174 

175 # Save camera settings 

176 self.camera_backup["reset"] = copy(self.camera.position) 

177 self.camera_backup["center"] = tuple(copy(center)) 

178 self.camera_backup["x_normal"] = [ 

179 center[0] - distance_from_center * box_mean_size, center[1], center[2] 

180 ] 

181 self.camera_backup["y_normal"] = [ 

182 center[0], center[1] - distance_from_center * box_mean_size, center[2] 

183 ] 

184 self.camera_backup["z_normal"] = [ 

185 center[0], center[1], center[2] - distance_from_center * box_mean_size 

186 ] 

187 

188 # Rescale axes helper 

189 self.axes_3d.scale = [self.camera.far] * 3 

190 

191 self.scene.add(self.point_clouds["main"]) 

192 self.scene.add(self.outline) 

193 self.scene.add(self.axticks) 

194 

195 # Update visibility of outline according to keyword arg 

196 self.outline.visible = self.show_outline 

197 self.axticks.visible = self.show_outline

199 def _create_point_cloud(self, positions): 

200 """ 

201 Make a point cloud using pythreejs 

202 """ 

203 self.points_geometry = p3.BufferGeometry( 

204 attributes={ 

205 'position': 

206 p3.BufferAttribute(array=positions), 

207 'color': 

208 p3.BufferAttribute( 

209 array=np.ones([positions.shape[0], 3], dtype='float32')) 

210 }) 

211 

212 pixel_ratio = config['plot']['pixel_ratio'] 

213 # Note that an additional factor of 2.5 (obtained from trial and error) seems to 

214 # be required to get the sizes right in the scene. 

215 self.points_material = p3.PointsMaterial(vertexColors='VertexColors', 

216 size=2.5 * self._pixel_size * 

217 pixel_ratio, 

218 transparent=True) 

219 return p3.Points(geometry=self.points_geometry, material=self.points_material)

242 def _generate_axis_ticks_and_labels(self, *, limits, box_size, components): 

243 """ 

244 Create ticklabels on outline edges 

245 """ 

246 if self.tick_size is None: 

247 self.tick_size = 0.05 * np.mean(box_size) 

248 ticks_and_labels = p3.Group() 

249 iden = np.identity(3, dtype=np.float32) 

250 ticker_ = ticker.MaxNLocator(5) 

251 lims = {x: limits[x] for x in "xyz"} 

252 

253 def get_offsets(dim, ind): 

254 if dim == 'x': 

255 return np.array([0, limits['y'][ind], limits['z'][ind]]) 

256 if dim == 'y': 

257 return np.array([limits['x'][ind], 0, limits['z'][ind]]) 

258 if dim == 'z': 

259 return np.array([limits['x'][ind], limits['y'][ind], 0]) 

260 

261 for axis, (x, dim) in enumerate(zip('xyz', components)): 

262 ticks = ticker_.tick_values(lims[x][0], lims[x][1]) 

263 for tick in ticks: 

264 if lims[x][0] <= tick <= lims[x][1]: 

265 tick_pos = iden[axis] * tick + get_offsets(x, 0) 

266 ticks_and_labels.add( 

267 self._make_axis_tick(string=value_to_string(tick, precision=1), 

268 position=tick_pos.tolist(), 

269 size=self.tick_size)) 

270 coord = components[dim] 

271 axis_label = f'{dim} [{coord.unit}]' if self.axlabels[ 

272 x] is None else self.axlabels[x] 

273 # Offset labels 5% beyond axis ticks to reduce overlap 

274 delta = 0.05 

275 ticks_and_labels.add( 

276 self._make_axis_tick(string=axis_label, 

277 position=(iden[axis] * 0.5 * np.sum(limits[x]) + 

278 (1.0 + delta) * get_offsets(x, 0) - 

279 delta * get_offsets(x, 1)).tolist(), 

280 size=self.tick_size * 0.3 * len(axis_label))) 

281 

282 return ticks_and_labels

284 def update_opacity(self, alpha): 

285 """ 

286 Update opacity of all points when opacity slider is changed. 

287 Take cut surface into account if present. 

288 

289 There is a strange effect with point clouds and opacities. 

290 Results are best when depthTest is False, at low opacities. 

291 But when opacities are high, the points appear in the order 

292 they were drawn, and not in the order they are with respect 

293 to the camera position. So for high opacities, we switch to 

294 depthTest = True. 

295 """ 

296 for key, a in alpha.items(): 

297 self.point_clouds[key].material.opacity = a 

298 self.point_clouds[key].material.depthTest = a > 0.5

311 def add_cut_surface(self, cut_surface_indices): 

312 """ 

313 Add a new opaque point cloud that represents the cut surface. 

314 This occurs either when we show the cut surface, or when we update the position 

315 of the cut surface (the point cloud is removed and added again at the new 

316 location). 

317 """ 

318 self.cut_surface_indices = cut_surface_indices 

319 self.point_clouds["cut"] = self._create_point_cloud( 

320 self.point_cloud_positions[cut_surface_indices]) 

321 self.point_clouds["cut"].geometry.attributes["color"].array = self.point_clouds[ 

322 "main"].geometry.attributes["color"].array[self.cut_surface_indices] 

323 self.scene.add(self.point_clouds["cut"])

331 def update_data(self, new_values, info=None): 

332 """ 

333 Update colors of points. 

334 """ 

335 array = new_values['data'] 

336 colors = self.scalar_map.to_rgba(array.values)[..., :3] 

337 self._unit = array.unit 

338 

339 if 'mask' in new_values: 

340 # We change the colors of the points in-place where masks are True 

341 masks_inds = np.where(new_values['mask'].values) 

342 masks_colors = self.masks_scalar_map.to_rgba( 

343 array.values[masks_inds])[..., :3] 

344 colors[masks_inds] = masks_colors 

345 

346 colors = colors.astype('float32') 

347 self.point_clouds["main"].geometry.attributes["color"].array = colors 

348 if "cut" in self.point_clouds: 

349 self.point_clouds["cut"].geometry.attributes["color"].array = colors[ 

350 self.cut_surface_indices]

361 def _update_colorbar(self): 

362 """ 

363 Create the colorbar figure and save it to png and update the image 

364 widget. 

365 Note that the figure is closed as soon as it is created to avoid it 

366 re-appearing further down the notebook. 

367 """ 

368 height_inches = self.figsize[1] / config['plot']['dpi'] 

369 cbar_fig = plt.figure(figsize=(height_inches * 0.2, height_inches), 

370 dpi=config['plot']['dpi']) 

371 cbar_ax = cbar_fig.add_axes([0.05, 0.02, 0.25, 0.94]) 

372 _ = ColorbarBase(cbar_ax, 

373 cmap=self.scalar_map.get_cmap(), 

374 norm=self.scalar_map.norm, 

375 extend=self.extend) 

376 if not isinstance(self.scalar_map.norm, LogNorm): 

377 _.formatter.set_useOffset(False) 

378 cbar_ax.set_ylabel(self._unit) 

379 # TODO If we set this position it is clipped somewhere. For now we 

380 # leave the default, which places unit to the right of the colorbar. 

381 # cbar_ax.yaxis.set_label_coords(-0.9, 0.5) 

382 self.cbar_image.value = fig_to_pngbytes(cbar_fig)

408 def camera_normal(self, position, ind): 

409 """ 

410 Move camera to requested normal, and flip if current position is equal 

411 to the requested position. 

412 """ 

413 if np.allclose(self.camera.position, position): 

414 position[ind] = 2.0 * self.camera_backup["center"][ind] - position[ind] 

415 self.move_camera(position=position)
